regimental_republic = {
	potential = {
		hidden_trigger = { 
			NOR = { 
				is_country_type = fallen_empire
				is_country_type = awakened_fallen_empire 
			} 		
		}
	}

	ethics = {
		AND = {
			has_ethic = "ethic_militarist"
			has_ethic = "ethic_individualist"
		}
		NOT = { has_ethic = "ethic_fanatic_individualist" }
		NOT = { has_ethic = "ethic_fanatic_militarist" }
	}
	
	modifier = {
		
		country_sector_cap_add = 3
		country_core_sector_planet_cap = -2
		country_leader_cap = 3
		navy_size_add = 12
		ship_hitpoints_mult = -0.05
		ship_upkeep_mult = -0.15
		army_morale = -0.10
		pop_fortification_defense = 0.15
		planet_fortification_strength = 0.33
		army_upkeep_mult = -0.15
	}
	
	election_term_years = 12
	election_term_variance = 2 # +/- 10 years
	election_type = oligarchic
	can_have_emergency_elections = yes
	emergency_election_cost = 300
	has_agendas = yes
	
	max_election_candidates = 4
	election_candidates = {
		modifier = {
			add = 1000
			OR = {
				leader_class = ruler
				AND = {
					exists = pop_faction
					pop_faction = {
						is_scope_valid = yes
						is_pop_faction_type = loyalist
					}
				}
			}
		}
		modifier = {
			add = 100
			OR = {
				leader_class = admiral
				leader_class = general
			}
		}
	}
		
	
	
	ruler_title = RT_CONSUL
}
